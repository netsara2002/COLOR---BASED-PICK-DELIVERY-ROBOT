#include <Wire.h>
#include <Adafruit_PWMServoDriver.h>

// --- ULTRASONIC SENSOR PINS ---
// Connect Sensor 1 to D2 (Trig) and D3 (Echo)
// Connect Sensor 2 to D4 (Trig) and D5 (Echo)
#define TRIG1_PIN 2
#define ECHO1_PIN 3
#define TRIG2_PIN 4
#define ECHO2_PIN 5

// Create PCA9685 object
Adafruit_PWMServoDriver pca = Adafruit_PWMServoDriver();

// Servo pulse width limits
#define MIN_PULSE_WIDTH  150   
#define MAX_PULSE_WIDTH  600   
#define SERVO_FREQ       50    

void setup() {
  Serial.begin(9600);
  Serial.println("Arduino Uno Robot Arm System Starting...");
  
  // -- Setup Sensors --
  pinMode(TRIG1_PIN, OUTPUT);
  pinMode(ECHO1_PIN, INPUT);
  pinMode(TRIG2_PIN, OUTPUT);
  pinMode(ECHO2_PIN, INPUT);
  
  // -- Initialize PCA9685 --
  pca.begin();
  pca.setPWMFreq(SERVO_FREQ);
  
  delay(1000);
  
  // -- Set initial positions --
  Serial.println("Moving to home position...");
  setServoAngle(0, 0);        // Base
  setServoAngle(1, 85);       // Shoulder 1
  setServoAngle(2, 180 - 85); // Shoulder 2 mirrored
  setServoAngle(3, 100);      // Elbow
  setServoAngle(4, 0);        // Wrist
  setServoAngle(6, 20);       // Gripper (open)
  
  delay(1000);
  Serial.println("READY. Waiting for BOTH sensors to see < 9cm...");
}

void loop() {
  // 1. Read Distance from Both Sensors
  long dist1 = getDistance(TRIG1_PIN, ECHO1_PIN);
  long dist2 = getDistance(TRIG2_PIN, ECHO2_PIN);

  // 2. Debugging: Print distances to Serial Monitor
  // This helps you see if sensors are working
  Serial.print("Sensor 1: "); Serial.print(dist1);
  Serial.print("cm  |  Sensor 2: "); Serial.print(dist2);
  Serial.println("cm");

  // 3. THE LOGIC CHECK
  // (dist > 0) check is added to filter out "0" errors if a sensor disconnects
  if ( (dist1 > 0 && dist1 < 9) && (dist2 > 0 && dist2 < 9) ) {
    
    Serial.println(">>> TRIGGERED! Both sensors < 9cm. Starting Sequence...");
    delay(1000); // Wait 1 second to ensure object is stable
    
    // Run your exact pick and place sequence
    runPickAndPlace();
    
    Serial.println(">>> Sequence Complete. Resetting and Waiting...");
    delay(2000); // Wait 2 seconds before scanning again
  }
  
  delay(100); // Small delay to stop serial monitor spamming too fast
}

// --- YOUR MOVEMENT SEQUENCE ---
void runPickAndPlace() {
  Serial.println("Starting pick and place sequence...");
  
  // 1. Base to 0° (starting position)
  Serial.println("1. Base to 0°");
  moveServoSmooth(0, 0, 0, 30);
  
  // 2. Open Gripper
  Serial.println("2. Opening gripper");
  moveServoSmooth(6, 20, 20, 30);
  
  // 3. Shoulders  (reach position)
  Serial.println("3. Shoulders to reach");
  moveShouldersOpposite(83, 83, 30);
  
  // 4. Elbow down  (reach down slowly)
  Serial.println("4. Elbow reaching down");
  moveServoSmooth(3, 100, 70, 30);
  
  // 5. Wrist  (orient gripper)
  Serial.println("5. Wrist positioning");
  moveServoSmooth(4, 0, 30, 30);
  
  delay(300);
  
  // 6. Close Gripper (grab object)
  Serial.println("6. GRABBING object");
  moveServoSmooth(6, 20, 110, 30);
  
  delay(500);
  
  // 7. Wrist to 10° (secure grip)
  Serial.println("7. Adjusting wrist");
  moveServoSmooth(4, 30, 0, 30);
  
  // 8. Elbow  (lift)
  Serial.println("8. Lifting object");
  moveServoSmooth(3, 70, 80, 30);
  
  // 9. Shoulders  (raise arm)
  Serial.println("9. Raising arm");
  moveShouldersOpposite(83, 110, 30);
  
  // 10. Base (rotate to drop position)
  Serial.println("10. Rotating to drop position");
  moveServoSmooth(0, 0, 150, 30);
  
  // 11. Wrist (position for release)
  Serial.println("11. Positioning wrist");
  moveServoSmooth(4, 0, 30, 30);
  
  // 12. Elbow stays 
  Serial.println("12. Maintaining elbow position");
  moveServoSmooth(3, 80, 80, 30);
  
  delay(300);
  
  // 13. Open Gripper (release object)
  Serial.println("13. RELEASING object");
  moveServoSmooth(6, 110, 20, 30);
  
  delay(500);
  
  // 14. Wrist up 
  Serial.println("14. Returning wrist");
  moveServoSmooth(4, 30, 0, 30);
  
  // 15. Base back to 0° 
  Serial.println("15. Returning to base home");
  moveServoSmooth(0, 150, 0, 30);

  // 16. Elbow home
  Serial.println("16. Returning to elbow home");
  moveServoSmooth(3, 80, 100, 30);

  // 17. Shoulder home
  Serial.println("17. Returning to start");
  moveShouldersOpposite(110, 83, 30);
}

// --- HELPER FUNCTIONS ---

// Function to read distance from ultrasonic sensor
long getDistance(int trigPin, int echoPin) {
  digitalWrite(trigPin, LOW);
  delayMicroseconds(2);
  digitalWrite(trigPin, HIGH);
  delayMicroseconds(10);
  digitalWrite(trigPin, LOW);
  
  long duration = pulseIn(echoPin, HIGH);
  long distance = duration * 0.034 / 2; // Convert to cm
  
  return distance;
}

// Convert angle (0-180) to PWM value
void setServoAngle(uint8_t channel, int angle) {
  angle = constrain(angle, 0, 180);
  int pulse = map(angle, 0, 180, MIN_PULSE_WIDTH, MAX_PULSE_WIDTH);
  pca.setPWM(channel, 0, pulse);
}

// Smooth servo movement
void moveServoSmooth(uint8_t channel, int startAngle, int endAngle, int stepDelay) {
  if (startAngle < endAngle) {
    for (int angle = startAngle; angle <= endAngle; angle++) {
      setServoAngle(channel, angle);
      delay(stepDelay);
    }
  } else {
    for (int angle = startAngle; angle >= endAngle; angle--) {
      setServoAngle(channel, angle);
      delay(stepDelay);
    }
  }
}

// Move mirrored shoulders together
void moveShouldersOpposite(int startAngle, int endAngle, int stepDelay) {
  if (startAngle < endAngle) {
    for (int angle = startAngle; angle <= endAngle; angle++) {
      setServoAngle(1, angle);         // Shoulder 1
      setServoAngle(2, 180 - angle);   // Shoulder 2 mirrored
      delay(stepDelay);
    }
  } else {
    for (int angle = startAngle; angle >= endAngle; angle--) {
      setServoAngle(1, angle);
      setServoAngle(2, 180 - angle);
      delay(stepDelay);
    }
  }
}
