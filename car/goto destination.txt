#include <Wire.h>
#include <LiquidCrystal_I2C.h>

// ==========================================
// PIN MAPPING (ESP32 SAFE MODE)
// ==========================================
LiquidCrystal_I2C lcd(0x27, 16, 2);

// IR Sensors (0 = BLACK)
const int s1 = 36;
const int s2 = 39;
const int s3 = 34;
const int s4 = 35;
const int s5 = 32;

// Motors (L298N)
const int lmb = 12, lmf = 14, lms = 13;
const int rmb = 26, rmf = 27, rms = 25;

// Color Sensor
const int color_s2 = 4;
const int color_s3 = 5;
const int color_out = 18;

// ==========================================
// SETTINGS
// ==========================================
int baseSpeed = 100;       // Speed for straight line
int turningSpeed = 180;    // Speed for turns

// Memory & Recovery (From your "Perfect" Code)
unsigned long lastDetectTime = 0;
unsigned long memoryDuration = 450;
String lastDirection = "straight";
bool turningToRecover = false;

// Mission Variables
int missionPhase = 1;   // 1=Line, 2=Turn, 3=Final Run
String targetColor = "NONE";
bool junctionLock = false;

// ==========================================
// SETUP
// ==========================================
void setup() {
  Serial.begin(115200);

  // Init Pins
  pinMode(s1, INPUT); pinMode(s2, INPUT); pinMode(s3, INPUT);
  pinMode(s4, INPUT); pinMode(s5, INPUT);

  pinMode(lmb, OUTPUT); pinMode(lmf, OUTPUT); pinMode(lms, OUTPUT);
  pinMode(rmb, OUTPUT); pinMode(rmf, OUTPUT); pinMode(rms, OUTPUT);

  pinMode(color_s2, OUTPUT); pinMode(color_s3, OUTPUT); pinMode(color_out, INPUT);

  // Init LCD
  lcd.init();
  lcd.backlight();

  // Color Detection Sequence
  lcd.print("Ready...");
  delay(1000);
  lcd.clear(); lcd.print("Scanning...");
  
  targetColor = detectColor();

  lcd.clear(); lcd.print("Target: "); lcd.print(targetColor);
  lcd.setCursor(0,1);
  if (targetColor == "RED") lcd.print("-> LEFT");
  else if (targetColor == "GREEN") lcd.print("-> RIGHT");
  else lcd.print("-> STRAIGHT");

  delay(2000);
  lcd.clear(); lcd.print("GO!");
}

// ==========================================
// MAIN LOOP
// ==========================================
void loop() {

  // -------- PHASE 1: FOLLOW LINE TO JUNCTION --------
  if (missionPhase == 1) {
    int ir1 = digitalRead(s1);
    int ir2 = digitalRead(s2);
    int ir3 = digitalRead(s3);
    int ir4 = digitalRead(s4);
    int ir5 = digitalRead(s5);

    // Check for Junction (All Black)
    if (!junctionLock && ir1==0 && ir2==0 && ir3==0 && ir4==0 && ir5==0) {
      stopMotors();
      delay(200);
      junctionLock = true;
      missionPhase = 2; // Go to Turn Phase
      return;
    }

    runLineFollower(); // Use the "Good" Logic
  }

  // -------- PHASE 2: EXECUTE TURN --------
  else if (missionPhase == 2) {
    executeTurn();
    missionPhase = 3; // Go to Final Run Phase
  }

  // -------- PHASE 3: FINAL RUN (FOLLOW LINE) & STOP --------
  else if (missionPhase == 3) {
    lcd.clear(); lcd.print("Arriving...");

    // HERE IS THE FIX:
    // Instead of delay(), we run the line follower in a loop.
    unsigned long startTime = millis();
    
    // Run for 600ms (Adjust this number for distance)
    while (millis() - startTime < 600) { 
       runLineFollower(); 
    }

    // After time is up, STOP FOREVER
    stopMotors();
    lcd.clear(); lcd.print("MISSION DONE");
    while (true); 
  }
}

// ==========================================
// THE "PERFECT" LINE FOLLOWER LOGIC
// ==========================================
void runLineFollower() {
  int ir1 = digitalRead(s1);
  int ir2 = digitalRead(s2);
  int ir3 = digitalRead(s3);
  int ir4 = digitalRead(s4);
  int ir5 = digitalRead(s5);

  bool allWhite = (ir1==1 && ir2==1 && ir3==1 && ir4==1 && ir5==1);

  // --- MEMORY UPDATE ---
  if (ir1==0 || ir2==0) { lastDirection = "left"; lastDetectTime = millis(); } 
  else if (ir4==0 || ir5==0) { lastDirection = "right"; lastDetectTime = millis(); }

  // --- RECOVERY (If line lost) ---
  if (allWhite) {
    if (millis() - lastDetectTime < memoryDuration) {
      if (!turningToRecover) { turningToRecover = true; }
      
      if (lastDirection == "left") { 
        leftMotorBackward(turningSpeed); rightMotorForward(turningSpeed); 
      } 
      else { 
        leftMotorForward(turningSpeed); rightMotorBackward(turningSpeed); 
      }
      return;
    } else {
      stopMotors();
      return;
    }
  }
  turningToRecover = false;

  // --- MOVEMENT LOGIC ---
  
  // Sharp Left
  if (ir1==0 && ir2==0 && ir3==0 && ir4==1 && ir5==1) { 
    leftMotorBackward(turningSpeed); rightMotorForward(turningSpeed); lastDirection = "left"; delay(50);
  }
  // Sharp Right
  else if (ir1==1 && ir2==1 && ir3==0 && ir4==0 && ir5==0) { 
    leftMotorForward(turningSpeed); rightMotorBackward(turningSpeed); lastDirection = "right"; delay(50);
  }
  // Center Forward
  else if (ir3==0) { 
    leftMotorForward(baseSpeed); rightMotorForward(baseSpeed); lastDirection = "straight";
  }
  // Slight Lefts
  else if (ir1==0 || ir2==0) { 
    leftMotorBackward(turningSpeed); rightMotorForward(turningSpeed); lastDirection = "left";
  }
  // Slight Rights
  else if (ir4==0 || ir5==0) { 
    leftMotorForward(turningSpeed); rightMotorBackward(turningSpeed); lastDirection = "right";
  }
  else {
    // Default Forward
    leftMotorForward(baseSpeed - 30); rightMotorForward(baseSpeed - 30);
  }
}

// ==========================================
// TURN LOGIC (Junction)
// ==========================================
void executeTurn() {
  if (targetColor == "RED") {
    lcd.clear(); lcd.print("TURN LEFT");
    leftMotorBackward(turningSpeed); rightMotorForward(turningSpeed);
    delay(500); // Blind turn time
  }
  else if (targetColor == "GREEN") {
    lcd.clear(); lcd.print("TURN RIGHT");
    leftMotorForward(turningSpeed); rightMotorBackward(turningSpeed);
    delay(500); // Blind turn time
  }
  else {
    // If no color (Straight), just push forward past junction
    leftMotorForward(baseSpeed); rightMotorForward(baseSpeed);
    delay(300);
    return;
  }

  // Find the line again
  unsigned long t0 = millis();
  while (millis() - t0 < 1500) {
    if (digitalRead(s3) == 0) break; // Line found!

    if (targetColor == "RED") {
      leftMotorBackward(160); rightMotorForward(160);
    } else {
      leftMotorForward(160); rightMotorBackward(160);
    }
  }
  stopMotors();
}

// ==========================================
// COLOR SENSOR
// ==========================================
String detectColor() {
  digitalWrite(color_s2, LOW); digitalWrite(color_s3, LOW);
  int r = pulseIn(color_out, LOW, 30000);

  digitalWrite(color_s2, HIGH); digitalWrite(color_s3, HIGH);
  int g = pulseIn(color_out, LOW, 30000);

  digitalWrite(color_s2, LOW); digitalWrite(color_s3, HIGH);
  int b = pulseIn(color_out, LOW, 30000);

  if (r < g && r < b && r < 1000) return "RED";
  if (g < r && g < b && g < 1000) return "GREEN";
  return "BLUE"; // Default/Straight
}

// ==========================================
// MOTOR HELPERS
// ==========================================
void leftMotorForward(int s) { digitalWrite(lmf, LOW); digitalWrite(lmb, HIGH); analogWrite(lms, s); }
void leftMotorBackward(int s){ digitalWrite(lmf, HIGH); digitalWrite(lmb, LOW); analogWrite(lms, s); }
void leftMotorStop()         { digitalWrite(lmf, LOW); digitalWrite(lmb, LOW); analogWrite(lms, 0); }

void rightMotorForward(int s){ digitalWrite(rmf, LOW); digitalWrite(rmb, HIGH); analogWrite(rms, s); }
void rightMotorBackward(int s){ digitalWrite(rmf, HIGH); digitalWrite(rmb, LOW); analogWrite(rms, s); }
void rightMotorStop()        { digitalWrite(rmf, LOW); digitalWrite(rmb, LOW); analogWrite(rms, 0); }

void stopMotors() { leftMotorStop(); rightMotorStop(); }
